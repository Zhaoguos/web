Listen 443
SSLStrictSNIVHostCheck off

SSLCipherSuite AESGCM:ALL:!DH:!EXPORT:!RC4:+HIGH:!MEDIUM:!LOW:!aNULL:!eNULL
SSLProtocol all -SSLv2 -SSLv3

<VirtualHost *:443>
DocumentRoot /www/wwwroot/web/dist/
ServerName www.shenbang1.com
ServerAlias *.shenbang1.com
#ErrorLog /var/logs/ssl_error.log

  <Directory /www/wwwroot/web/dist/>
      Options FollowSymLinks ExecCGI
      AllowOverride All
      Order allow,deny
      Allow from all
      #Require all granted	# For 2.4.x only
  </Directory>

SSLEngine on
SSLCertificateFile /www/server/apache/conf/sslkey/shenbang1_com_apache/shenbang1_com.crt
SSLCertificateKeyFile /www/server/apache/conf/sslkey/shenbang1_com_apache/shenbang1_com.key
SSLCertificateChainFile /www/server/apache/conf/sslkey/shenbang1_com_apache/comodo.crt

</VirtualHost>
<VirtualHost www.domain.com:443>
    DocumentRoot "/var/www/html"
    ServerName www.domain.com
    SSLEngine on
    SSLCertificateFile /usr/local/apache/conf/2_www.domain.com_cert.crt
    SSLCertificateKeyFile /usr/local/apache/conf/3_www.domain.com.key
    SSLCertificateChainFile /usr/local/apache/conf/1_root_bundle.crt
</VirtualHost>
